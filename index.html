var N='OBJ_data.json';
function doGet(e){
  var f=DriveApp.getFilesByName(N);
  var data=f.hasNext()?f.next().getBlob().getDataAsString():JSON.stringify({status:'empty'});
  return ContentService.createTextOutput(data).setMimeType(ContentService.MimeType.JSON);
}
function doPost(e){
  var f=DriveApp.getFilesByName(N);
  if(f.hasNext())f.next().setContent(e.postData.contents);
  else DriveApp.createFile(N,e.postData.contents,MimeType.PLAIN_TEXT);
  return ContentService.createTextOutput(JSON.stringify({status:'ok'})).setMimeType(ContentService.MimeType.JSON);
}
